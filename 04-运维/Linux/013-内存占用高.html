# 内存占用高

# 查看内存使用，以G为单位

$ free -g
1

# 查看内存使用，以M为单位

$ free -m
1

# 内存占用高的前20

$ ps aux | head -1;ps aux |grep -v PID |sort -rn -k +4 | head -20
1

# 查看进程的内存占用

$ pidstat -r -p 24427 1 5
1
